    <script>

        // Data storage with localStorage persistence

        let employees = [];

        let workRequests = [];

        let motelRequests = [];

        let carRequests = [];

        let currentId = 1;

       

        // Storage management functions

        function saveToStorage() {

            try {

                const data = {

                    employees: employees,

                    workRequests: workRequests,

                    motelRequests: motelRequests,

                    carRequests: carRequests,

                    currentId: currentId

                };

                localStorage.setItem('employeeManagementData', JSON.stringify(data));

            } catch (error) {

                console.error('Error saving to storage:', error);

                showNotification('Error saving data to storage', 'error');

            }

        }

       

        function loadFromStorage() {

            try {

                const savedData = localStorage.getItem('employeeManagementData');

                if (savedData) {

                    const data = JSON.parse(savedData);

                    employees = data.employees || [];

                    workRequests = data.workRequests || [];

                    motelRequests = data.motelRequests || [];

                    carRequests = data.carRequests || [];

                    currentId = data.currentId || 1;

                    return true;

                }

            } catch (error) {

                console.error('Error loading from storage:', error);

                showNotification('Error loading data from storage', 'error');

            }

            return false;

        }

       

        function clearStorage() {

            if (confirm('Are you sure you want to clear all data? This action cannot be undone.')) {

                localStorage.removeItem('employeeManagementData');

                employees = [];

                workRequests = [];

                motelRequests = [];

                carRequests = [];

                currentId = 1;

                updateAllDisplays();

                showNotification('All data cleared successfully', 'success');

            }

        }

       

        function exportData() {

            const data = {

                employees: employees,

                workRequests: workRequests,

                motelRequests: motelRequests,

                carRequests: carRequests,

                exportDate: new Date().toISOString()

            };

           

            const dataStr = JSON.stringify(data, null, 2);

            const dataBlob = new Blob([dataStr], {type: 'application/json'});

            const url = URL.createObjectURL(dataBlob);

           

            const link = document.createElement('a');

            link.href = url;

            link.download = `employee_data_${new Date().toISOString().split('T')[0]}.json`;

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

            URL.revokeObjectURL(url);

           

            showNotification('Data exported successfully!');

        }

       

        function updateAllDisplays() {

            updateEmployeeList();

            updateEmployeeDropdowns();

            updateWorkRequestsList();

            updateMotelRequestsList();

            updateCarRequestsList();

            updatePendingApprovals();

        }

       

        // Tab navigation

        function showTab(tabName) {

            // Hide all tabs

            document.querySelectorAll('.tab-content').forEach(tab => {

                tab.classList.remove('active');

            });

            document.querySelectorAll('.nav-tab').forEach(tab => {

                tab.classList.remove('active');

            });

           

            // Show selected tab

            document.getElementById(tabName).classList.add('active');

            event.target.classList.add('active');

        }

       

        // Notification system

        function showNotification(message, type = 'success') {

            const notification = document.getElementById('notification');

            notification.textContent = message;

            notification.className = `notification ${type} show`;

           

            setTimeout(() => {

                notification.classList.remove('show');

            }, 3000);

        }

        document.getElementById('employeeForm').addEventListener('submit', function(e) {

            e.preventDefault();

           

            const employee = {

                id: document.getElementById('empId').value,

                name: document.getElementById('empName').value,

        // Approval management

        function updatePendingApprovals() {

            const list = document.getElementById('pendingApprovals');

            list.innerHTML = '';

           

            const pendingWork = workRequests.filter(req => req.status === 'pending');

            const pendingMotel = motelRequests.filter(req => req.status === 'pending');

            const pendingCar = carRequests.filter(req => req.status === 'pending');

           

            [...pendingWork, ...pendingMotel, ...pendingCar].forEach(req => {

                const reqCard = document.createElement('div');

                reqCard.className = 'card';

               

                let type = 'Work';

                if (motelRequests.includes(req)) type = 'Motel';

                if (carRequests.includes(req)) type = 'Car Rental';

               

                reqCard.innerHTML = `

                    <h4>${type} Request #${req.id}</h4>

                    <p><strong>Employee:</strong> ${req.employeeName}</p>

                    <p><strong>Submitted:</strong> ${req.submittedAt}</p>

                    ${type === 'Work' ? `<p><strong>Location:</strong> ${req.location}</p>` : ''}

                    ${type === 'Motel' ? `<p><strong>Motel:</strong> ${req.motelName}, ${req.city}</p>` : ''}

                    ${type === 'Car Rental' ? `<p><strong>Car Type:</strong> ${req.carType}</p>` : ''}

                    <div style="margin-top: 15px;">

                        <button class="btn btn-success" onclick="approveRequest(${req.id}, '${type.toLowerCase().replace(' ', '')}')">Approve</button>

                        <button class="btn btn-danger" onclick="rejectRequest(${req.id}, '${type.toLowerCase().replace(' ', '')}')">Reject</button>

                    </div>

                `;

                list.appendChild(reqCard);

            });

           

            if (pendingWork.length === 0 && pendingMotel.length === 0 && pendingCar.length === 0) {

                list.innerHTML = '<p style="text-align: center; color: #666; font-style: italic;">No pending approvals</p>';

            }

        }

       

        function approveRequest(id, type) {

            let requests;

            switch(type) {

                case 'work':

                    requests = workRequests;

                    break;

                case 'motel':

                    requests = motelRequests;

                    break;

                case 'carrental':

                    requests = carRequests;

                    break;

            }

           

            const request = requests.find(req => req.id === id);

            if (request) {

                request.status = 'approved';

                request.approvedAt = new Date().toLocaleString();

                saveToStorage(); // Save to localStorage

                updatePendingApprovals();

                updateWorkRequestsList();

                updateMotelRequestsList();

                updateCarRequestsList();

                showNotification(`${type.charAt(0).toUpperCase() + type.slice(1)} request approved!`);

            }

        }

       

        function rejectRequest(id, type) {

            let requests;

            switch(type) {

                case 'work':

                    requests = workRequests;

                    break;

                case 'motel':

                    requests = motelRequests;

                    break;

                case 'carrental':

                    requests = carRequests;

                    break;

            }

           

            const request = requests.find(req => req.id === id);

            if (request) {

                request.status = 'rejected';

                request.rejectedAt = new Date().toLocaleString();

                saveToStorage(); // Save to localStorage

                updatePendingApprovals();

                updateWorkRequestsList();

                updateMotelRequestsList();

                updateCarRequestsList();

                showNotification(`${type.charAt(0).toUpperCase() + type.slice(1)} request rejected!`, 'error');

            }

        }
